üü¢ What is Propositional Logic?

Propositional logic, also known as Boolean logic, is a formal system where statements (called propositions) are either true (1) or false (0).

from itertools import product

def truth_table(variables, expression):
    print(f"{' '.join(variables)} | {expression}")
    print("-" * 20)
    
    for values in product([0, 1], repeat=len(variables)):
        env = dict(zip(variables, values))
        print("  ".join(str(env[var]) for var in variables), "|", eval(expression, {}, env))

truth_table(["X", "Y"], "X or Y")
